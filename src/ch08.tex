            \begin{itemize}
                \item 已知 $A_i = 1 + 2 + 3 + ... + i$，$S_n = A_1 + A_2 + A_3 + ... + A_n$。\\

                    $A_3 = 1 + 2 + 3 = 6$\\
                    $S_3 = A_1 + A_2 + A_3 = 1 + 3 + 6 = 10$

            \end{itemize}

        \end{exampleblock}
